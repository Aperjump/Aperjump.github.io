使用metaprogramming编程系统能够执行我们所写的代码生成新的代码，这样可以减少开销，某些用户自定义的计算可以在compile期间进行，能够在性能和接口的简单性上带来好处。

```
//meta/pow3.hpp

template<int N>
class Pow3 {
public:
        enum { result = 3 * Pow3<N-1>::result };
};

template<>
class Pow3<0> {
public:
        enum { result = 1 };
};

#endif
```
```
// meta/sqrt1.hpp

#ifndef SQRT_HPP
#define SQRT_HPP

template<int N, int LO = 0, int HI = N>
class Sqrt {
public:
        enum {mid = (LO + HI + 1) / 2};
        enum {result = (N < mid * mid) ? Sqrt<N,LO,mid-1>::result : Sqrt<N,mid,HI>::result};
};

template<int N, int M>
class Sqrt<N,M,M> {
public:
        enum {result = M};
}

#endif
```
但是这样的写法会导致`？：`两侧的模板都被实例化，从而大大增加了模板实例化工作量。
```
// meta/sqrt1.hpp

#ifndef SQRT_HPP
#define SQRT_HPP

template<int N, int LO = 0, int HI = N>
class Sqrt {
public:
        enum {mid = (LO + HI + 1) / 2};
        enum {result = (N < mid * mid) ? Sqrt<N,LO,mid-1>::result : Sqrt<N,mid,HI>::result};
};

template<int N, int M>
class Sqrt<N,M,M> {
public:
        enum {result = M};
}

#endif
```
```
//meta/sqrt2.hpp

#include "ifthenelse.hpp"

template<int N, int LO = 0, int HI = N>
class Sqrt {
public:
        enum { mid = (LO + HI + 1) / 2 };
        typedef typename IfThenElse<(N < mid * mid), Sqrt<N, LO, mid-1>,
                        Sqrt<N,mid,HI>>::ResultT SubT;
        enum {result = SubT::result};
};

template<int N, int S>
class Sqrt<N,S,S> {
public:
        enum { result = S };
};
```
定义一个`typedef`的时候并不会导致C++编译器实例化该实体。

## 1. 归纳变量
```
//meta/sqrt4.hpp

#ifndef SQRT_HPP
#define SQRT_HPP

#include "ifthenelse.hpp"

template<int N>
class Value {
public:
        enum { result = N };
};

template<int N, int I = 0>
class Sqrt {
public:
        typedef typename IfThenElse<(I * I < N), Sqrt<N, I+1>,
                        Value<I>>::ResultT SubT;
        enum { result = SubT::result };
};

#endif
```
**c++标准建议最多只进行17层递归实例化**
实际开发中很少会使用template meta program，但是有时他们可以藏在普通模板的内部，用于实现那些对性能要求很严格的算法，从而大大提高效率。
```
//meta/loop2.hpp

#ifndef LOOP_HPP
#define LOOP_HPP

template <int DIM, typename T>
class DotProduct {
public:
        static T result (T* a, T* b) {
                return *a * *b + DotProduct<DIM-1,T>::result(a+1,b+1);
        }
};

template<typename T>
class DotProduct<1,T> {
public:
        static T result (T* a, T* b) {
                return *a * *b;
        }
};

#endif
```
