## 1. 辨别基本类型
首先需要开发一个用于辨别某个类型是否为基本类型的模板。我们一方面假设一个类型不是基本类型，另一方面需要特化基本类型。
```
//types/type1.hpp

template <typename T>
class IsFundaT {
public:
        enum { Yes = 0, No = 1 };
};

// 特化基本类型的宏
#define MK_FUNDA_TYPE(T)
        template<> class IsFundaT<T> {
        public:
                enum { Yes = 1, No = 0 };
        };

MK_FUNDA_TYPE(void)

MK_FUNDA_TYPE(bool)
MK_FUNDA_TYPE(char)
MK_FUNDA_TYPE(signed char)
MK_FUNDA_TYPE(unsigned char)
MK_FUNDA_TYPE(wchar_t)
MK_FUNDA_TYPE(signed short)
MK_FUNDA_TYPE(unsigned short)
MK_FUNDA_TYPE(signed int)
MK_FUNDA_TYPE(unsigned int)
MK_FUNDA_TYPE(signed long)
MK_FUNDA_TYPE(unsigned long)
#if LONGLONG_EXISTS
        MK_FUNDA_TYPE(signed long long)
        MK_FUNDA_TYPE(unsigned long long)
#endif

MK_FUNDA_TYPE(float)
MK_FUNDA_TYPE(double)
MK_FUNDA_TYPE(long double)

#undef MK_FUNDA_TYPE
```
## 2. 组合类型
组合类型时构造自其他类型的类型，包括普通类型，指针类型，引用类型，数组类型.
```
//types/type2.hpp

template<typename T>
class CompoundT {
public:
        enum { IsPtrT = 0, IsRefT = 0, IsArrayT = 0,
                   IsFuncT = 0, IsPtrMemT = 0 };
        typedef T BaseT;
        typedef T BottomT;
        typedef CompoundT<void> ClassT;
};

template<typename T>
class CompoundT<T&> {
public:
        enum { IsPtrT = 0, IsRefT = 1, IsArrayT = 0,
                IsFuncT = 0, IsPtrMemT = 0 };
        typedef T BaseT;
        typedef typename Compound<T>::BottomT BottomT;
        typedef CompoundT<void> ClassT;
};

template<typename T>
class CompoundT<T*> {
public:
        enum { IsPtrT = 1, IsRefT = 0, IsArrayT = 0,
                IsFuncT = 0, IsPtrMemT = 0 };
        typedef T BaseT;
        typedef typename CompoundT<T>::BottomT BottomT;
        typedef CompoundT<void> ClassT;
};

template<typename T>
class CompoundT<T[N]> {
public:
        enum { IsPtT = 0, IsRefT = 0, IsArrayT = 1,
                IsFuncT = 0, IsPtrMemT = 0 };

        typedef typename CompoundT<T>::BottomT BottomT;
        tyepdef CompoundT<void> ClassT;
```

## 3. 函数类型辨别
函数参数不确定
SFINAE(substitution-failuere-is-not-an-error): 找到一种构造，这个构造对函数类型时无效的，但是对他类型是有效的或者相反。
```
template <typename T>
class IsFunctionT {
private:
   typedef char One;
   typedef struct { char a[2]; } Two;
   template<typename U> static One test(...);
   template<typename U> static Two test(U (*)[1]); // This test overloading I cannot understand 
public:
   enum { Yes = sizeof(IsFunctionT<T>::test<T>(0) == 1};
   enum { No = !Yes };
};
```
还需要对引用类型和void类型进行特化，在此掠过。
```
/ types/type3.hpp
template<typename T>
class IsFunctionT {
private:
        typedef char One;
        typedef struct { char a[2]; } Two;
        template<typename U> static One test(...);
        template<typename U> static Two test(U (*)[1]);
public:
        enum { Yes = sizeof(IsFunctionT<T>::test<T>(0)) == 1 };
        enum { No = !Yes };
};

template<typename T>
class IsFunctionT<T&> {
public:
        enum { Yes = 0 };
        enum { No = !Yes };
};

template<>
class IsFunctionT<void> {
public:
        enum { Yes = 0 };
        enum { No = !Yes };
};

template<>
class IsFunctionT<void const> {
public:
        enum { Yes = 0 };
        enum { No = !Yes };
};

template<typename T>
class CompoundT {
public:
        enum { IsPtrT = 0, IsRefT = 0, IsArrayT = 0,
                IsFuncT = IsFunctionT<T>::Yes, IsPtrMemT = 0 };
        typedef T BaseT;
        typedef T BottomT;
        typedef CompoundT<void> ClassT;
};
```
